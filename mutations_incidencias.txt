/**
 * EJEMPLOS DE MUTACIONES PARA INCIDENCIAS - KANBAN
 * Actualizado según nuevo modelo de BD (MODELO_BD_NUEVO.sql)
 */

# 1. Crear Tipos de Incidencia (catálogo previo)
mutation CrearTipoIncidencia {
  createTipoIncidencia(nombre_tipo: "Falla de hardware") {
    id_tipo_incidencia
    nombre_tipo
  }
}

# 2. Crear Nueva Incidencia (Modal inicial)
# - id_usuario_genera_reporte: tomado del context.user (usuario autenticado que crea el reporte)
# - id_usuario_reporta: el usuario final que reporta la falla (puede ser diferente)
# - id_tipo_incidencia: FK al catálogo de tipos
# - prioridad: 'Baja' | 'Media' | 'Alta' | 'Crítica'  (default: 'Media')
# - El estatus queda en "Pendiente" por defecto
mutation CrearIncidencia {
  createIncidencia(
    id_bien: "XXXXX-XXXXX-XXXXX-XXXXX"
    id_usuario_reporta: 5
    id_tipo_incidencia: 1
    descripcion_falla: "Error de calibración en la bandeja de papel 2"
    prioridad: "Alta"
    unidad: "Hospital General Regional"
  ) {
    id_incidencia
    prioridad
    estatus_reparacion
    tipoIncidencia {
      nombre_tipo
    }
    usuarioGeneraReporte {
      nombre_completo
    }
    usuarioReporta {
      nombre_completo
    }
  }
}

# 3. Asignar la Incidencia a un Técnico (Opcional, antes o en cualquier momento)
mutation AsignarTecnico {
  asignarIncidencia(
    id_incidencia: "1"
    id_usuario_asignado: 4 # ID del usuario técnico
  ) {
    id_incidencia
    usuarioAsignado {
      nombre_completo
    }
  }
}

# 4. Mover la Incidencia a "En proceso" y agregar la primera nota de revisión
#  - Se usa cuando alguien arrastra la tarjeta a "En proceso"
mutation EmpezarRevision {
  pasarAEnProceso(
    id_incidencia: "1"
    contenido_nota: "Se procedió a revisar la bandeja 2, parece atorada."
  ) {
    id_incidencia
    estatus_reparacion
    notas {
      contenido_nota
    }
  }
}

# 5. Agregar una nota adicional de seguimiento
mutation AgregarNota {
  agregarNotaSeguimiento(
    id_incidencia: "1"
    contenido_nota: "Reemplazado el cable de rodillos, reensamblando."
  ) {
    id_nota
    contenido_nota
    fecha_creacion
  }
}

# 6. Resolver Incidencia (Modal/Sección final al mover a Resuelto o Cerrar)
# - Puede ser "Resuelto", "Cerrado" o "Sin resolver"
mutation FinalizarIncidencia {
  resolverIncidencia(
    id_incidencia: "1"
    estatus_cierre: "Resuelto"
    resolucion_textual: "Se cambió pieza defectuosa y se calibraron rodillos. Exitosa."
  ) {
    id_incidencia
    estatus_reparacion
    resolucion_textual
    fecha_resolucion
    usuarioResuelve {
      nombre_completo
    }
  }
}

# 7. Cambiar estatus genérico o manualmente (Retorno a pendiente, etc.)
mutation CambiarStatus {
  updateIncidenciaEstatus(
    id_incidencia: "1"
    estatus_reparacion: "Pendiente"
  ) {
    id_incidencia
    estatus_reparacion
  }
}
